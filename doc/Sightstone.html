def initialize(api_key, region='euw')
  @api_key = api_key
  @region = region
  
  @summoner = SummonerModule.new(self)
  @champion = ChampionModule.new(self)
  @game = GameModule.new(self)
  @league = LeagueModule.new(self)
  @stats = StatsModule.new(self)
  @team = TeamModule.new(self)
end
